# AttrMasking Requirements

# Environment setup:
#   conda create -n attrmasking_env python=3.8
#   conda activate attrmasking_env
#
# Step 1 — Install RDKit:
#   conda install -c conda-forge rdkit
#
# Step 2 — Install DGL (CPU version):
#   pip install dgl -f https://data.dgl.ai/wheels/torch-2.1/cpu/repo.html
#
# Step 3 — Install dgllife:
#   pip install dgllife
#
# Step 4 — Install DeepPurpose:
#   pip install git+https://github.com/bp-kelley/descriptastorus
#   pip install DeepPurpose
#
# Step 5 — Install remaining dependencies:
#   pip install -r requirements.txt
#
# NOTE: Pre-trained AttrMasking weights (~200MB) are downloaded automatically
# on first run from the dgllife model zoo.
