# AttentiveFP Requirements
# -------------------------
# AttentiveFP shares the same conda environment as AttrMasking.
# If you already have attrmasking_env set up, just activate it.
#
# Environment setup (if starting fresh):
#   conda create -n attrmasking_env python=3.8
#   conda activate attrmasking_env
#
# Step 1 — Install Pytorch/RDKit:
#   pip install torch
#   conda install -c conda-forge rdkit
#
# Step 2 — Install DGL:
#   pip install dgl -f https://data.dgl.ai/wheels/torch-2.1/cu126/repo.html
#
# Step 3 — Install dgllife:
#   pip install dgllife
#
# Step 4 — Install DeepPurpose:
#   pip install descriptastorus
#   pip install DeepPurpose
#   pip install pydantic
#
# Step 5 — Install remaining dependencies:
#   pip install -r requirements.txt
#
# NOTE: Unlike AttrMasking, AttentiveFP does NOT need pre-trained weights.
# It trains from scratch using graph attention on each task.

PyTDC
numpy
pandas
